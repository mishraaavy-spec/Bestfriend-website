    <script>
        // --- THREE.JS BACKGROUND ---
        let scene, camera, renderer, hearts = [];
        const heartCount = 25;

        function initThree() {
            scene = new THREE.Scene();
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('three-canvas'), alpha: true, antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);

            // Light
            const light = new THREE.PointLight(0xffffff, 1.5);
            light.position.set(5, 5, 5);
            scene.add(light);
            scene.add(new THREE.AmbientLight(0xffffff, 0.5));

            // Heart Shape
            const x = 0, y = 0;
            const heartShape = new THREE.Shape();
            heartShape.moveTo(x + 5, y + 5);
            heartShape.bezierCurveTo(x + 5, y + 5, x + 4, y, x, y);
            heartShape.bezierCurveTo(x - 6, y, x - 6, y + 7, x - 6, y + 7);
            heartShape.bezierCurveTo(x - 6, y + 11, x - 3, y + 15.4, x + 5, y + 19);
            heartShape.bezierCurveTo(x + 12, y + 15.4, x + 16, y + 11, x + 16, y + 7);
            heartShape.bezierCurveTo(x + 16, y + 7, x + 16, y, x + 10, y);
            heartShape.bezierCurveTo(x + 7, y, x + 5, y + 5, x + 5, y + 5);

            const geometry = new THREE.ExtrudeGeometry(heartShape, { depth: 2, bevelEnabled: true, bevelThickness: 1, bevelSize: 1 });
            const material = new THREE.MeshPhongMaterial({ color: 0xec4899, shininess: 100 });

            for (let i = 0; i < heartCount; i++) {
                const mesh = new THREE.Mesh(geometry, material.clone());
                mesh.position.set(
                    (Math.random() - 0.5) * 80,
                    (Math.random() - 0.5) * 80,
                    (Math.random() - 0.5) * 40 - 20
                );
                mesh.rotation.x = Math.PI;
                mesh.rotation.z = Math.random() * Math.PI;
                const scale = Math.random() * 0.15 + 0.05;
                mesh.scale.set(scale, scale, scale);
                
                // Add movement data
                mesh.userData = {
                    offset: Math.random() * 100,
                    speed: 0.005 + Math.random() * 0.01
                };

                scene.add(mesh);
                hearts.push(mesh);
            }

            camera.position.z = 50;
        }

        function animateThree() {
            requestAnimationFrame(animateThree);
            hearts.forEach(heart => {
                heart.position.y += Math.sin(Date.now() * 0.001 + heart.userData.offset) * 0.05;
                heart.rotation.y += 0.01;
            });
            renderer.render(scene, camera);
        }

        // --- NAVIGATION LOGIC ---
        function navigate(step) {
            const current = document.querySelector('.active-card');
            const next = document.getElementById(`step${step}`);
            const progress = (step / 5) * 100;

            document.getElementById('progressFill').style.width = progress + '%';

            gsap.to(current, {
                opacity: 0,
                y: -40,
                scale: 0.9,
                duration: 0.6,
                ease: "power2.inOut",
                onComplete: () => {
                    current.classList.remove('active-card');
                    next.classList.add('active-card');
                    
                    gsap.fromTo(next, 
                        { opacity: 0, y: 40, scale: 0.9 },
                        { opacity: 1, y: 0, scale: 1, duration: 0.8, ease: "power3.out" }
                    );

                    // Trigger internal staggers
                    gsap.from(next.querySelectorAll('h1, p, button, .bento-item'), {
                        opacity: 0,
                        y: 20,
                        stagger: 0.1,
                        duration: 0.5,
                        delay: 0.2
                    });
                }
            });
        }

        // --- POLAROID PARALLAX ---
        document.addEventListener('mousemove', (e) => {
            const polaroid = document.getElementById('polaroid');
            if (!polaroid || !document.getElementById('step4').classList.contains('active-card')) return;
            
            const x = (window.innerWidth / 2 - e.pageX) / 30;
            const y = (window.innerHeight / 2 - e.pageY) / 30;
            
            gsap.to(polaroid, {
                rotationY: x,
                rotationX: -y,
                duration: 0.5,
                ease: "power2.out"
            });
        });

        // --- FINALE ---
        function runFinale() {
            const btn = document.getElementById('finalBtn');
            const text = document.getElementById('finalText');

            gsap.to(btn, { opacity: 0, scale: 0.5, duration: 0.4, onComplete: () => btn.style.display = 'none' });
            
            setTimeout(() => {
                text.classList.remove('hidden');
                gsap.fromTo(text, { opacity: 0, y: 30 }, { opacity: 1, y: 0, duration: 1, ease: "back.out" });
                
                // Confetti Sequence
                const end = Date.now() + (5 * 1000);

                (function frame() {
                    confetti({
                        particleCount: 3,
                        angle: 60,
                        spread: 55,
                        origin: { x: 0, y: 0.8 },
                        colors: ['#ec4899', '#ef4444', '#ffd1da']
                    });
                    confetti({
                        particleCount: 3,
                        angle: 120,
                        spread: 55,
                        origin: { x: 1, y: 0.8 },
                        colors: ['#ec4899', '#ef4444', '#ffd1da']
                    });

                    if (Date.now() < end) {
                        requestAnimationFrame(frame);
                    }
                }());

                // Emoji Burst
                confetti({
                    particleCount: 40,
                    spread: 70,
                    origin: { y: 0.6 },
                    shapes: ['circle'],
                    scalar: 2,
                    content: ['❤️', '💖', '✨', '🎀']
                });

                // Hearts Swirl Away
                hearts.forEach((heart, i) => {
                    gsap.to(heart.position, {
                        y: 100,
                        x: (Math.random() - 0.5) * 200,
                        z: 50,
                        duration: 3,
                        delay: i * 0.05,
                        ease: "power2.in"
                    });
                    gsap.to(heart.material, { opacity: 0, duration: 2, delay: i * 0.05 });
                });
            }, 500);
        }

        // Initialize everything
        window.onload = () => {
            initThree();
            animateThree();
            
            // Initial Animation
            gsap.fromTo(".active-card", 
                { opacity: 0, y: 50 }, 
                { opacity: 1, y: 0, duration: 1.2, ease: "power4.out" }
            );
        };

        // Resize handler
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
    </script>
